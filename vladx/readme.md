# Учебник по языку VladX

VladX — это интерпретируемый язык программирования с русским синтаксисом. Он прост в VladX и позволяет писать код на русском языке.

## 1. Базовые типы данных

В VladX есть следующие типы данных:

```vladx
# Числа
пусть число = 42
пусть дробное = 3.14

# Строки
пусть имя = "Мария"

# Логические значения
пусть истина = истина
пусть ложь = ложь

# Ничто (null/ничего)
пусть пусто = nothing
```

Вывод значений:

```vladx
печать(число)      # 42
печать(имя)        # Мария
печать(истина)     # истина
печать(ложь)       # ложь
печать(пусто)      # nothing
```

## 2. Переменные и константы

### Переменные (могут изменяться)

```vladx
пусть счетчик = 0
печать(счетчик)    # 0

счетчик = 1
печать(счетчик)    # 1

счетчик = счетчик + 1
печать(счетчик)    # 2
```

### Константы (не могут изменяться)

```vladx
конст ПИ = 3.14159
печать(ПИ)         # 3.14159

# Попытка изменить константу вызовет ошибку!
# ПИ = 10           # Ошибка!
```

## 3. Арифметические операции

```vladx
пусть а = 10
пусть б = 3

печать(а + б)      # 13  (сложение)
печать(а - б)      # 7   (вычитание)
печать(а * б)      # 30  (умножение)
печать(а / б)      # 3.333... (деление)
печать(а % б)      # 1   (остаток от деления)

# Комбинированные операции
а = а + 5          # а = 15
а = а - 2          # а = 13
а = а * 2          # а = 26
```

## 4. Операции сравнения

```vladx
пусть а = 10
пусть б = 20

печать(а == б)     # ложь  (равно)
печать(а != б)     # истина (не равно)
печать(а < б)      # истина (меньше)
печать(а > б)      # ложь  (больше)
печать(а <= б)     # истина (меньше или равно)
печать(а >= б)     # ложь  (больше или равно)
```

## 5. Логические операции

```vladx
пусть а = истина
пусть б = ложь

печать(а и б)      # ложь  (И - оба должны быть истина)
печать(а или б)    # истина (ИЛИ - хотя бы один истина)
печать(не а)       # ложь  (НЕ - инверсия)
```

## 6. Условные конструкции

### Простое условие `если`

```vladx
пусть число = 10

если (число > 5) {
    печать("Число больше 5")
}
```

### Условие с альтернативой `если ... иначе`

```vladx
пусть число = 3

если (число > 5) {
    печать("Число больше 5")
} иначе {
    печать("Число меньше или равно 5")
}
```

### Множественные условия

```vladx
пусть оценка = 4

если (оценка == 5) {
    печать("Отлично!")
} иначеесли (оценка == 4) {
    печать("Хорошо")
} иначеесли (оценка == 3) {
    печать("Удовлетворительно")
} иначе {
    печать("Нужно постараться")
}
```

## 7. Циклы

### Цикл `пока` (while)

```vladx
пусть счетчик = 0

пока (счетчик < 3) {
    печать("Счетчик: " + счетчик)
    счетчик = счетчик + 1
}
# Выведет: Счетчик: 0, Счетчик: 1, Счетчик: 2
```

### Цикл `для` (for)

```vladx
# Простой цикл от 1 до 5
для (пусть i = 1; i <= 5; i = i + 1) {
    печать(i)
}
# Выведет: 1, 2, 3, 4, 5

# Вычисление суммы
пусть сумма = 0
для (пусть i = 1; i <= 10; i = i + 1) {
    сумма = сумма + i
}
печать("Сумма 1-10: " + сумма)  # 55
```

### Управление циклом

```vladx
# Прервать цикл
для (пусть i = 1; истина; i = i + 1) {
    если (i > 5) {
        прервать          # Выход из цикла
    }
    печать(i)
}

# Перейти к следующей итерации
для (пусть i = 1; i <= 5; i = i + 1) {
    если (i == 3) {
        продолжить       # Пропустить текущую итерацию
    }
    печать(i)
}
# Выведет: 1, 2, 4, 5
```

## 8. Функции

### Создание функции

```vladx
функция привет(имя) {
    вернуть "Привет, " + имя + "!"
}

печать(привет("Мария"))     # Привет, Мария!
печать(привет("Алексей"))   # Привет, Алексей!
```

### Функции с несколькими параметрами

```vladx
функция сложить(а, б) {
    вернуть а + б
}

печать(сложить(5, 3))       # 8
печать(сложить(10, 20))     # 30
```

### Функции без параметров

```vladx
функция сейчас() {
    вернуть "Сейчас время: 12:00"
}

печать(сейчас())            # Сейчас время: 12:00
```

### Локальные переменные

```vladx
функция вычислить(а, б) {
    пусть сумма = а + б    # Локальная переменная
    пусть произведение = а * б
    вернуть сумма + произведение
}

печать(вычислить(2, 3))     # 11 (2+3 + 2*3)
# переменные сумма и произведение недоступны здесь
```

## 9. Объекты

### Создание объекта

```vladx
пусть человек = {
    имя: "Иван",
    возраст: 25,
    город: "Москва"
}

печать(человек)
# { имя: Иван, возраст: 25, город: Москва }
```

### Доступ к свойствам

```vladx
печать(человек.имя)         # Иван
печать(человек.возраст)     # 25
```

### Изменение свойств

```vladx
человек.возраст = 26
печать(человек.возраст)     # 26

человек.город = "Санкт-Петербург"
печать(человек.город)       # Санкт-Петербург
```

### Добавление новых свойств

```vladx
человек.профессия = "Программист"
печать(человек.профессия)   # Программист
```

### Вложенные объекты

```vladx
пусть автомобиль = {
    марка: "Toyota",
    характеристики: {
        мощность: 150,
        год: 2020,
        цвет: "белый"
    }
}

печать(автомобиль.характеристики.мощность)  # 150
печать(автомобиль.харакстики.год)          # 2020
```

### Объекты с функциями

```vladx
пусть калькулятор = {
    сложить: функция(а, б) {
        вернуть а + б
    },
    умножить: функция(а, б) {
        вернуть а * б
    }
}

печать(калькулятор.сложить(5, 3))     # 8
печать(калькулятор.умножить(4, 7))    # 28
```

## 10. Массивы

### Создание массива

```vladx
пусть числа = [1, 2, 3, 4, 5]
пусть имена = ["Маша", "Саша", "Даша"]
пусть смешанный = [1, "два", истина, nothing]
```

### Доступ к элементам

```vladx
печать(числа[0])           # 1 (первый элемент)
печать(имена[1])           # Саша
печать(числа[4])           # 5
```

### Изменение элементов

```vladx
числа[0] = 10
печать(числа[0])           # 10

имена[1] = "Гриша"
печать(имена)              # [Маша, Гриша, Даша]
```

### Длина массива

```vladx
печать(длина(числа))       # 5
```

### Перебор массива

```vladx
пусть числа = [1, 2, 3, 4, 5]
пусть сумма = 0

для (пусть i = 0; i < длина(числа); i = i + 1) {
    сумма = сумма + числа[i]
}

печать("Сумма: " + сумма)  # 15
```

## 11. Практические примеры

### Калькулятор

```vladx
функция калькулятор(а, б, операция) {
    если (операция == "+") {
        вернуть а + б
    } иначеесли (операция == "-") {
        вернуть а - б
    } иначеесли (операция == "*") {
        вернуть а * б
    } иначеесли (операция == "/") {
        вернуть а / б
    } иначе {
        вернуть "Неизвестная операция"
    }
}

печать(калькулятор(10, 5, "+"))   # 15
печать(калькулятор(10, 5, "*"))   # 50
```

### Факториал

```vladx
функция факториал(н) {
    если (н <= 1) {
        вернуть 1
    }
    вернуть н * факториал(н - 1)
}

печать(факториал(5))    # 120
печать(факториал(7))    # 5040
```

### Поиск максимума в массиве

```vladx
функция максимум(массив) {
    пусть макс = массив[0]
    для (пусть i = 1; i < длина(массив); i = i + 1) {
        если (массив[i] > макс) {
            макс = массив[i]
        }
    }
    вернуть макс
}

пусть числа = [3, 7, 2, 9, 1, 5]
печать(максимум(числа))     # 9
```

### Библиотека

```vladx
пусть математика = {
    квадрат: функция(х) {
        вернуть х * х
    },
    куб: функция(х) {
        вернуть х * х * х
    },
    абс: функция(х) {
        если (х < 0) {
            вернуть -х
        }
        вернуть х
    }
}

печать(математика.квадрат(5))   # 25
печать(математика.куб(3))       # 27
печать(математика.абс(-7))      # 7
```

## 12. Встроенные функции

```vladx
# Математические
печать(максимум(1, 5, 3))          # 5
печать(минимум(1, 5, 3))           # 1
печать(корень(16))                 # 4
печать(округлить(3.7))             # 4
печать(пол(3.7))                   # 3
печать(потолок(3.2))               # 4

# Ожидание (в миллисекундах)
ждать(1000)                        # Пауза на 1 секунду

# Текущее время
печать(время())                    # Метка времени
```
