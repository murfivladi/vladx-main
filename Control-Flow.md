# Управляющие конструкции

## Если (if/else)

### Синтаксис

```vlad
если условие {
    // код выполнится если условие истина
}
```

### Если-иначе (if-else)

```vlad
если условие {
    // код если условие истина
} иначе {
    // код если условие ложь
}
```

### Если-иначеесли-иначе (if-elseif-else)

```vlad
если условие1 {
    // код если условие1 истина
} иначеесли условие2 {
    // код если условие2 истина
} иначе {
    // код если все условия ложь
}
```

### Примеры

```vlad
пусть возраст = 25

если возраст >= 18 {
    печать("Вы совершеннолетний")
} иначе {
    печать("Вы несовершеннолетний")
}
```

```vlad
пусть оценка = 85

если оценка >= 90 {
    печать("Отлично!")
} иначеесли оценка >= 75 {
    печать("Хорошо!")
} иначеесли оценка >= 60 {
    печать("Удовлетворительно")
} иначе {
    печать("Нужно подтянуть")
}
```

## Пока (while)

### Синтаксис

```vlad
пока условие {
    // код выполняется пока условие истина
}
```

### Примеры

```vlad
пусть счетчик = 0

пока счетчик < 5 {
    печать(счетчик)
    счетчик = счетчик + 1
}
```

### Бесконечный цикл

```vlad
пока истина {
    // этот цикл будет выполняться вечно
    прервать // выход из цикла
}
```

## Для (for)

### Синтаксис

```vlad
для (инициализация; условие; обновление) {
    // код выполняется пока условие истина
}
```

### Примеры

```vlad
// Простой цикл
для (пусть i = 0; i < 10; i = i + 1) {
    печать(i)
}
```

```vlad
// Цикл в обратном порядке
для (пусть i = 10; i > 0; i = i - 1) {
    печать(i)
}
```

```vlad
// Использование массива
пусть числа = [1, 2, 3, 4, 5]
пусть сумма = 0

для (пусть i = 0; i < длина(числа); i = i + 1) {
    сумма = сумма + числа[i]
}

печать("Сумма: " + сумма)
```

## Прервать (break)

Оператор `прервать` / `break` немедленно выходит из цикла.

### Пример

```vlad
для (пусть i = 0; i < 10; i = i + 1) {
    если i == 5 {
        прервать // выход из цикла при i == 5
    }
    печать(i)
}
// Выведет: 0, 1, 2, 3, 4
```

## Продолжить (continue)

Оператор `продолжить` / `continue` пропускает текущую итерацию цикла и переходит к следующей.

### Пример

```vlad
для (пусть i = 0; i < 10; i = i + 1) {
    если i % 2 == 0 {
        продолжить // пропустить четные числа
    }
    печать(i)
}
// Выведет: 1, 3, 5, 7, 9
```

## Переключатель (switch)

### Синтаксис

```vlad
выбор (выражение) {
    когда значение1 {
        // код если выражение == значение1
    }
    когда значение2 {
        // код если выражение == значение2
    }
    поумолчанию {
        // код если ни одно значение не совпало
    }
}
```

### Примеры

```vlad
пусть день = "Понедельник"

выбор (день) {
    когда "Понедельник" {
        печать("Начало рабочей недели")
    }
    когда "Пятница" {
        печать("Почти выходные!")
    }
    когда "Суббота" {
        печать("Выходной!")
    }
    когда "Воскресенье" {
        печать("Выходной!")
    }
    поумолчанию {
        печать("Обычный рабочий день")
    }
}
```

## Попытка-перехват (try-catch-finally)

### Синтаксис

```vlad
попытка {
    // код который может вызвать ошибку
} перехват (ошибка) {
    // код для обработки ошибки
} наконец {
    // код который выполнится в любом случае
}
```

### Примеры

```vlad
попытка {
    пусть результат = 10 / 0
} перехват (ошибка) {
    печать("Произошла ошибка: " + ошибка)
} наконец {
    печать("Блок finally выполнен")
}
```

```vlad
попытка {
    пусть текст = "Не число"
    пусть число = Number(текст)
    если typeof число != "number" {
        бросить("Не удалось преобразовать в число")
    }
} перехват (ошибка) {
    печать("Ошибка: " + ошибка)
}
```

## Бросить исключение (throw)

### Синтаксис

```vlad
бросить(значение_ошибки)
```

### Примеры

```vlad
функция проверитьВозраст(возраст) {
    если возраст < 0 {
        бросить("Возраст не может быть отрицательным!")
    }
    если возраст > 150 {
        бросить("Возраст слишком большой!")
    }
    вернуть истина
}

попытка {
    проверитьВозраст(-5)
} перехват (ошибка) {
    печать("Ошибка: " + ошибка)
}
```

## Вложенные конструкции

### Пример

```vlad
пусть числа = [[1, 2], [3, 4], [5, 6]]

для (пусть i = 0; i < длина(числа); i = i + 1) {
    для (пусть j = 0; j < длина(числа[i]); j = j + 1) {
        если числа[i][j] % 2 == 0 {
            печать("Четное число: " + числа[i][j])
        } иначе {
            печать("Нечетное число: " + числа[i][j])
        }
    }
}
```

## Практические примеры

### Поиск в массиве

```vlad
пусть числа = [1, 3, 5, 7, 9]
пусть цель = 5
пусть найдено = ложь

для (пусть i = 0; i < длина(числа); i = i + 1) {
    если числа[i] == цель {
        найдено = истина
        прервать
    }
}

если найдено {
    печать("Число " + цель + " найдено!")
} иначе {
    печать("Число " + цель + " не найдено")
}
```

### Сумма элементов массива

```vlad
пусть числа = [1, 2, 3, 4, 5]
пусть сумма = 0

для (пусть i = 0; i < длина(числа); i = i + 1) {
    сумма = сумма + числа[i]
}

печать("Сумма: " + сумма)
```

### Факториал

```vlad
пусть число = 5
пусть факториал = 1

для (пусть i = 1; i <= число; i = i + 1) {
    факториал = факториал * i
}

печать("Факториал " + число + ": " + факториал)
```
